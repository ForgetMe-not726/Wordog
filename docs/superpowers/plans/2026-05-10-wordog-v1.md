> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

```typescript
// src/lib/auth.ts
import NextAuth from "next-auth";
import Credentials from "next-auth/providers/credentials";
import { compare } from "bcryptjs";
import { prisma } from "./prisma";

export const { handlers, auth, signIn, signOut } = NextAuth({
  providers: [
    Credentials({
      credentials: {
        email: { label: "Email", type: "email" },
        password: { label: "Password", type: "password" },
      },
      async authorize(credentials) {
        if (!credentials?.email || !credentials?.password) return null;
        const user = await prisma.user.findUnique({
          where: { email: credentials.email as string },
        });
        if (!user) return null;
        const isValid = await compare(credentials.password as string, user.passwordHash!);
        if (!isValid) return null;
        return { id: user.id, email: user.email, name: user.name };
      },
    }),
  ],
  pages: {
    signIn: "/auth/login",
  },
  callbacks: {
    async session({ session, token }) {
      if (session.user) {
        session.user.id = token.sub!;
      }
      return session;
    },
  },
});
```

```typescript
// src/app/api/auth/register/route.ts
import { NextResponse } from "next/server";
import { hash } from "bcryptjs";
import { prisma } from "@/lib/prisma";

export async function POST(request: Request) {
  try {
    const { name, email, password, dogName } = await request.json();

    if (!email || !password || !dogName) {
      return NextResponse.json({ error: "请填写所有必填字段" }, { status: 400 });
    }

    const existing = await prisma.user.findUnique({ where: { email } });
    if (existing) {
      return NextResponse.json({ error: "该邮箱已注册" }, { status: 400 });
    }

    const passwordHash = await hash(password, 12);
    const defaultBreed = await prisma.dogBreed.findFirst({ where: { isDefault: true } });

    const user = await prisma.user.create({
      data: {
        name,
        email,
        passwordHash,
        dog: {
          create: {
            name: dogName,
            breedId: defaultBreed!.id,
          },
        },
      },
    });

    return NextResponse.json({ userId: user.id }, { status: 201 });
  } catch {
    return NextResponse.json({ error: "服务器错误" }, { status: 500 });
  }
}
```

```typescript
// src/lib/ebbinghaus.ts

export const REVIEW_INTERVALS = [0, 1, 2, 4, 7, 15, 30] as const;

export function getNextReviewStage(currentStage: number): number {
  const next = currentStage + 1;
  const maxStageIndex = REVIEW_INTERVALS.length - 1;
  if (next > maxStageIndex) {
    // Word is fully mastered, no more review needed
    return -1;
  }
  return next;
}

export function getNextReviewDate(stage: number): Date {
  const days = REVIEW_INTERVALS[stage];
  const date = new Date();
  date.setHours(0, 0, 0, 0);
  date.setDate(date.getDate() + days);
  return date;
}

export function getDueReviewsCount(totalStages: { reviewStage: number; nextReviewAt: Date | null }[]): number {
  const now = new Date();
  return totalStages.filter((w) => {
    if (w.reviewStage === 0 || w.nextReviewAt === null) return false;
    return w.nextReviewAt <= now;
  }).length;
}
```

```tsx
"use client";

import { useState, useEffect, useCallback } from "react";
import { useRouter } from "next/navigation";
import MeaningChoice from "@/components/learn/MeaningChoice";
import KnowJudge from "@/components/learn/KnowJudge";
import WordCard from "@/components/ui/WordCard";

interface WordData {
  id: string;
  word: string;
  phonetic: string | null;
  meaning: string;
  example: string | null;
  synonyms: string[];
  antonyms: string[];
  confusables: string[];
  derivatives: string[];
  options: string[];
}

type Round = "meaning" | "know1" | "know2";

interface QueueItem {
  word: WordData;
  round: Round;
}

export default function NewWordPage() {
  const router = useRouter();
  const [session, setSession] = useState<QueueItem[]>([]);
  const [currentIndex, setCurrentIndex] = useState(0);
  const [showCard, setShowCard] = useState(false);
  const [cardWord, setCardWord] = useState<WordData | null>(null);
  const [cardCorrect, setCardCorrect] = useState(true);
  const [done, setDone] = useState(false);

  // Fetch words for a new session
  useEffect(() => {
    fetch("/api/learn", { method: "POST" })
      .then((r) => r.json())
      .then((words: WordData[]) => {
        const queue: QueueItem[] = words.map((w) => ({ word: w, round: "meaning" }));
        setSession(queue);
      });
  }, []);

  const recordAndAdvance = useCallback(
    async (wordId: string, round: number, correct: boolean) => {
      await fetch("/api/learn", {
        method: "PUT",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({ wordId, round, correct }),
      });
    },
    [],
  );

  const handleMeaningAnswer = useCallback(
    (selected: string) => {
      const item = session[currentIndex];
      if (!item) return;
      const correct = selected === item.word.meaning;
      setCardWord(item.word);
      setCardCorrect(correct);
      setShowCard(true);

      recordAndAdvance(item.word.id, 1, correct);

      if (!correct) {
        // Insert into queue 5 positions later
        const newSession = [...session];
        const removed = newSession.splice(currentIndex, 1)[0];
        const insertAt = Math.min(currentIndex + 5, newSession.length);
        newSession.splice(insertAt, 0, removed);
        setSession(newSession);
        // Don't advance index since we removed the current one
      }
    },
    [session, currentIndex, recordAndAdvance],
  );

  const handleKnowJudge = useCallback(
    (know: boolean) => {
      const item = session[currentIndex];
      if (!item) return;
      const roundNum = item.round === "know1" ? 2 : 3;
      setCardWord(item.word);
      setCardCorrect(know);
      setShowCard(true);

      recordAndAdvance(item.word.id, roundNum, know);

      if (!know) {
        // Fall back to meaning round
        const newSession = [...session];
        newSession[currentIndex] = { ...item, round: "meaning" };
        setSession(newSession);
      }
    },
    [session, currentIndex, recordAndAdvance],
  );

  const handleCardDismiss = useCallback(() => {
    setShowCard(false);
    setCardWord(null);

    const item = session[currentIndex];
    if (!item) return;

    if (item.round === "meaning" && cardCorrect) {
      // Move to know1
      const newSession = [...session];
      newSession[currentIndex] = { ...item, round: "know1" };
      setSession(newSession);
    } else if (item.round === "know1" && cardCorrect) {
      // Move to know2
      const newSession = [...session];
      newSession[currentIndex] = { ...item, round: "know2" };
      setSession(newSession);
    } else if (item.round === "know2" && cardCorrect) {
      // Done with this word
      const newSession = [...session];
      newSession.splice(currentIndex, 1);
      setSession(newSession);
      if (newSession.length === 0) {
        setDone(true);
      }
    } else {
      // Wrong answer, advance to next word (current stays as meaning for retry)
      setCurrentIndex((i) => Math.min(i + 1, session.length - 1));
    }
  }, [session, currentIndex, cardCorrect]);

  if (done) {
    return (
      <div className="min-h-screen flex items-center justify-center bg-green-50 p-4">
        <div className="text-center space-y-4">
          <p className="text-4xl">🎉</p>
          <p className="text-xl font-bold text-gray-800">今日新词学习完成</p>
          <button
            onClick={() => router.push("/learn")}
            className="bg-green-500 text-white rounded-xl px-8 py-3 font-bold"
          >
            返回
          </button>
        </div>
      </div>
    );
  }

  if (session.length === 0) {
    return (
      <div className="min-h-screen flex items-center justify-center bg-green-50">
        <p className="text-gray-400">加载中...</p>
      </div>
    );
  }

  const current = session[currentIndex];
  if (!current) {
    return (
      <div className="min-h-screen flex items-center justify-center bg-green-50">
        <p className="text-gray-400">加载中...</p>
      </div>
    );
  }

  return (
    <div className="min-h-screen bg-green-50 p-4 flex items-center justify-center">
      <div className="w-full max-w-md">
        <p className="text-center text-sm text-gray-400 mb-4">
          进度 {currentIndex + 1}/{session.length}
          {current.round === "meaning" ? " · 选词意" : current.round === "know1" ? " · 认识判断(1/2)" : " · 认识判断(2/2)"}
        </p>

        {current.round === "meaning" ? (
          <MeaningChoice
            word={current.word.word}
            phonetic={current.word.phonetic}
            example={current.word.example}
            options={current.word.options}
            onAnswer={handleMeaningAnswer}
          />
        ) : (
          <KnowJudge word={current.word.word} onJudge={handleKnowJudge} />
        )}
      </div>

      {showCard && cardWord && (
        <WordCard
          word={cardWord.word}
          phonetic={cardWord.phonetic}
          meaning={cardWord.meaning}
          example={cardWord.example}
          synonyms={cardWord.synonyms}
          antonyms={cardWord.antonyms}
          confusables={cardWord.confusables}
          derivatives={cardWord.derivatives}
          correct={cardCorrect}
          onDismiss={handleCardDismiss}
```

```typescript
// src/app/api/learn/route.ts
import { NextResponse } from "next/server";
import { auth } from "@/lib/auth";
import { prisma } from "@/lib/prisma";
import { getNextReviewStage, getNextReviewDate } from "@/lib/ebbinghaus";

const SESSION_SIZE = 10;

function shuffle<T>(arr: T[]): T[] {
  const a = [...arr];
  for (let i = a.length - 1; i > 0; i--) {
    const j = Math.floor(Math.random() * (i + 1));
    [a[i], a[j]] = [a[j], a[i]];
  }
  return a;
}

// POST: Start a new learning session
export async function POST() {
  const session = await auth();
  if (!session?.user?.id) return NextResponse.json({ error: "Unauthorized" }, { status: 401 });

  const needMeaningRound = await prisma.userWord.findMany({
    where: {
      userId: session.user.id,
      round: 1,
      status: "learning",
    },
    take: SESSION_SIZE,
    include: { word: true },
  });

  if (needMeaningRound.length >= SESSION_SIZE) {
    return NextResponse.json(formatWords(needMeaningRound.map((uw) => uw.word)));
  }

  const existingWordIds = needMeaningRound.map((uw) => uw.wordId);
  const newWords = await prisma.word.findMany({
    where: {
      id: { notIn: existingWordIds.length > 0 ? existingWordIds : undefined },
      userWords: { none: { userId: session.user.id } },
    },
    take: SESSION_SIZE - needMeaningRound.length,
  });

  const allWords = [
    ...needMeaningRound.map((uw) => uw.word),
    ...newWords,
  ];

  const upserts = newWords.map((w) =>
    prisma.userWord.create({
      data: {
        userId: session.user.id,
        wordId: w.id,
        round: 1,
        status: "learning",
      },
    }),
  );
  await Promise.all(upserts);

  return NextResponse.json(formatWords(allWords));
}

// PUT: Submit a learning result
export async function PUT(request: Request) {
  const session = await auth();
  if (!session?.user?.id) return NextResponse.json({ error: "Unauthorized" }, { status: 401 });

  const { wordId, round, correct } = await request.json();

  const userWord = await prisma.userWord.findUnique({
    where: { userId_wordId: { userId: session.user.id, wordId } },
  });

  if (!userWord) return NextResponse.json({ error: "Not found" }, { status: 404 });

  // Record study log
  await prisma.studyRecord.create({
    data: { userId: session.user.id, wordId, round, correct },
  });

  if (round === 3 && correct) {
    // Word is learned, enter review queue
    const nextStage = getNextReviewStage(0);
    await prisma.userWord.update({
      where: { id: userWord.id },
      data: {
        status: "review",
        reviewStage: nextStage === -1 ? 0 : nextStage,
        nextReviewAt: nextStage === -1 ? null : getNextReviewDate(nextStage),
        learnedAt: new Date(),
        round: 0,
      },
    });

    // Award food currency
    await prisma.dog.updateMany({
      where: { userId: session.user.id },
      data: { foodCurrency: { increment: 10 } },
    });
  } else if (!correct && (round === 2 || round === 3)) {
    // Reset to meaning round
    await prisma.userWord.update({
      where: { id: userWord.id },
      data: { round: 1 },
    });
  } else if (round === 1) {
    await prisma.userWord.update({
      where: { id: userWord.id },
      data: { round: correct ? 2 : 1 },
    });
  }

  return NextResponse.json({ success: true });
}

function formatWords(words: { id: string; word: string; phonetic: string | null; meaning: string; example: string | null; synonyms: string | null; antonyms: string | null; confusables: string | null; derivatives: string | null }[]) {
  return shuffle(
    words.map((w) => {
      const otherWords = shuffle(
        words.filter((o) => o.id !== w.id && o.meaning !== w.meaning),
      ).slice(0, 3);

      return {
        id: w.id,
        word: w.word,
        phonetic: w.phonetic,
        meaning: w.meaning,
        example: w.example,
        synonyms: w.synonyms ? JSON.parse(w.synonyms) : [],
        antonyms: w.antonyms ? JSON.parse(w.antonyms) : [],
        confusables: w.confusables ? JSON.parse(w.confusables) : [],
        derivatives: w.derivatives ? JSON.parse(w.derivatives) : [],
        options: shuffle([w.meaning, ...otherWords.map((o) => o.meaning)]),
      };
    }),
  );
}
```

```typescript
export function computeMoodChange(
  currentStreak: number,

    change -= 10; // Hunger penalty
  }
  if (currentStreak === 0) {
    change -= 10; // Missed a day
  }
  return change;
}

export function computeFullnessConsumption(lastFedAt: Date): number {
  const now = new Date();
  const daysSince = Math.floor((now.getTime() - lastFedAt.getTime()) / (1000 * 60 * 60 * 24));
  return Math.min(daysSince * DAILY_FOOD_CONSUMPTION, FULLNESS_MAX);
}

export function computeStreak(checkInDates: Date[]): number {
  if (checkInDates.length === 0) return 0;
  const sorted = [...checkInDates].sort((a, b) => b.getTime() - a.getTime());
  const today = new Date();
  today.setHours(0, 0, 0, 0);

  const latest = new Date(sorted[0]);
  latest.setHours(0, 0, 0, 0);

  if (latest.getTime() !== today.getTime() && latest.getTime() !== today.getTime() - 86400000) {
    return 0;
  }

  let streak = 1;
  for (let i = 1; i < sorted.length; i++) {
    const prev = new Date(sorted[i - 1]);
    const curr = new Date(sorted[i]);
    prev.setHours(0, 0, 0, 0);
    curr.setHours(0, 0, 0, 0);
    const diff = (prev.getTime() - curr.getTime()) / 86400000;
    if (diff === 1) {
      streak++;
    } else {
      break;
    }
  }
  return streak;
}

export const FOOD_REWARDS: Record<string, number> = {

  complete_review_batch: 5,
```

```typescript
import { auth } from "@/lib/auth";
import { prisma } from "@/lib/prisma";

// GET: Get dog state (with daily update)
export async function GET() {
  const session = await auth();
  if (!session?.user?.id) return NextResponse.json({ error: "Unauthorized" }, { status: 401 });

  const dog = await prisma.dog.findUnique({
    where: { userId: session.user.id },
    include: {
      breed: true,
      accessories: {
        where: { equipped: true },
        include: { accessory: true },
      },
    },
  });

  if (!dog) return NextResponse.json({ error: "No dog" }, { status: 404 });

  const checkIns = await prisma.checkIn.findMany({
    where: { userId: session.user.id },
    select: { date: true },
    orderBy: { date: "desc" },
  });
  const streak = computeStreak(checkIns.map((c) => c.date));

  // Apply daily consumption
  const consumption = computeFullnessConsumption(dog.createdAt);
  const newFullness = Math.max(0, dog.fullness - consumption);
  const moodChange = computeMoodChange(streak, newFullness);
  const newMood = Math.max(0, Math.min(100, dog.mood + moodChange));

  if (newFullness !== dog.fullness || newMood !== dog.mood) {
    await prisma.dog.update({
      where: { id: dog.id },
      data: { fullness: newFullness, mood: newMood },
    });
  }

  return NextResponse.json({
    id: dog.id,
    name: dog.name,
    breed: dog.breed,
    foodCurrency: dog.foodCurrency,
    fullness: newFullness,
    mood: newMood,
    equippedAccessories: dog.accessories.map((da) => da.accessory),
    streak,
  });
}

// PUT: Feed / buy accessory / unlock breed
export async function PUT(request: Request) {
  const session = await auth();
  if (!session?.user?.id) return NextResponse.json({ error: "Unauthorized" }, { status: 401 });

  const { action, accessoryId, breedId } = await request.json();
  const dog = await prisma.dog.findUnique({ where: { userId: session.user.id } });
  if (!dog) return NextResponse.json({ error: "No dog" }, { status: 404 });

  if (action === "feed") {
    if (dog.foodCurrency < 10) return NextResponse.json({ error: "狗粮不足" }, { status: 400 });
    const updated = await prisma.dog.update({
      where: { id: dog.id },
      data: {
        foodCurrency: dog.foodCurrency - 10,
        fullness: Math.min(100, dog.fullness + 30),
      },
    });
    return NextResponse.json(updated);
  }

  if (action === "buy_accessory" && accessoryId) {
    const accessory = await prisma.accessory.findUnique({ where: { id: accessoryId } });
    if (!accessory) return NextResponse.json({ error: "Not found" }, { status: 404 });
    if (dog.foodCurrency < accessory.price) return NextResponse.json({ error: "狗粮不足" }, { status: 400 });

    const existing = await prisma.dogAccessory.findUnique({
      where: { dogId_accessoryId: { dogId: dog.id, accessoryId } },
    });
    if (existing) return NextResponse.json({ error: "已拥有" }, { status: 400 });

    await prisma.dogAccessory.create({
      data: { dogId: dog.id, accessoryId },
    });
    const updated = await prisma.dog.update({
      where: { id: dog.id },
      data: { foodCurrency: dog.foodCurrency - accessory.price },
    });
    return NextResponse.json(updated);
  }

  if (action === "unlock_breed" && breedId) {
    const breed = await prisma.dogBreed.findUnique({ where: { id: breedId } });
    if (!breed) return NextResponse.json({ error: "Not found" }, { status: 404 });
    if (dog.foodCurrency < breed.unlockCost) return NextResponse.json({ error: "狗粮不足" }, { status: 400 });

    const updated = await prisma.dog.update({
      where: { id: dog.id },
      data: { breedId: breed.id, foodCurrency: dog.foodCurrency - breed.unlockCost },
      include: { breed: true },
    });
    return NextResponse.json(updated);
  }

  return NextResponse.json({ error: "Invalid action" }, { status: 400 });
}
```
